Configuration

entity = "http://dbpedia.org/resource/Aristotle"  # Target entity URI

same_as = "No"                                    # Only for LODsyndesis ("Yes"/"No")

source = "DBpedia"  # Choose: "LODsyndesis" or "DBpedia"
